function readDB(name) {
  const p = path.join(DATA_DIR, `${name}.json`);
  if (!fs.existsSync(p)) return {};
  try { return JSON.parse(fs.readFileSync(p, 'utf8')); } catch { return {}; }
}
function writeDB(name, data) {
  fs.writeFileSync(path.join(DATA_DIR, `${name}.json`), JSON.stringify(data, null, 2));
}

  if (!message.content.startsWith(prefix)) return;
  const args = message.content.slice(prefix.length).trim().split(/ +/);

}

// Filter & anti-spam
const userMsgMap = new Map();
async function handleAntiSpam(msg) { 
  if (isMod(msg.member)) return;
  const cfg = readDB('config').antispam; 
  if (!cfg?.enabled) return; 
  const key = msg.author.id, now = Date.now(); 
  if (!userMsgMap.has(key)) userMsgMap.set(key, []); 
  const msgs = userMsgMap.get(key).filter(t => now - t < cfg.interval); 
  msgs.push(now); 
  userMsgMap.set(key, msgs); 
  if (msgs.length >= cfg.threshold) { 
    if (cfg.action === 'delete') await msg.delete().catch(() => {}); 
    else if (cfg.action === 'mute') await msg.member?.timeout(10 * 60 * 1000, 'Auto-spam').catch(() => {}); 
    userMsgMap.delete(key); 
  } 
}
const abuseCache = new Map();
async function handleAntiAbuse(msg) { 
  if (isMod(msg.member)) return;
  const cfg = readDB('config').antiabuse; 
  if (!cfg?.enabled) return; 
  if (msg.mentions.users.size >= 5 && cfg.blockMassPing) { 
    await msg.delete().catch(() => {}); 
    await msg.member?.timeout(5 * 60 * 1000, 'Mass mention'); 
  } 
  const actions = ['ban', 'kick']; 
  for (const a of actions) if (msg.content.toLowerCase().includes(`/${a}`) && cfg[`blockMass${a.charAt(0).toUpperCase() + a.slice(1)}`]) { 
    const key = `${msg.author.id}_${a}`, now = Date.now(); 
    if (!abuseCache.has(key)) abuseCache.set(key, []); 
    const times = abuseCache.get(key).filter(t => now - t < 10000); 
    times.push(now); 
    abuseCache.set(key, times); 
    if (times.length >= 3) { 
      await msg.member?.timeout(60 * 60 * 1000, 'Mass mod command spam').catch(() => {}); 
      abuseCache.delete(key); 
    } 
  } 
}

function filterCheck(content, userId) { 
  const lists = readDB('lists'); 
  const wl = lists.whitelist?.[userId]; 
  if (wl?.scope === 'all') return { matched: false }; 
  const db = readDB('wordfilter'); 
  const words = db.words || []; 
  const anti = readDB('config').antibypass !== false; 
  let normalized = content.toLowerCase(); 
  if (anti) normalized = normalized.replace(/[^\w\s]/g, '').replace(/([a-z])\1{2,}/g, '$1$1'); 
  for (const w of words) { 
    if (wl?.words?.includes(w)) continue; 
    if (normalized.includes(w)) return { matched: true, word: w }; 
  } 
  return { matched: false }; 
}

        }
        if (!versionHash) continue;
        let timestamp = rawDate ? new Date(rawDate).getTime() : Date.now();
        if (isNaN(timestamp)) timestamp = Date.now();
        allUpdates.push({ source: site, versionHash: String(versionHash).trim(), platform: String(platform).trim(), timestamp, downloadNumbers: numbers || null, downloadUrl: downloadUrl || null, liveMsg: liveMsg || cfg.liveMessage || 'Live update detected!', patchNote: patchNote || cfg.patchNote || 'Salt is patched.' });
      } catch (e) { console.warn(`⚠️ Failed to fetch ${site}:`, e.message); }
    }
    if (allUpdates.length === 0) return;
    allUpdates.sort((a, b) => b.timestamp - a.timestamp);
    latestUpdate = allUpdates[0];
    const stored = readDB('robloxState') || {};
    let hasNewUpdate = false;
    for (const update of allUpdates) { if (stored.lastHash !== update.versionHash) { hasNewUpdate = true; latestUpdate = update; break; } }
    if (!hasNewUpdate && stored.lastHash === latestUpdate.versionHash) return;
    stored.lastHash = latestUpdate.versionHash;
    stored.lastChecked = Date.now();
    stored.lastUpdateData = latestUpdate;
    stored.updateHistory = stored.updateHistory || [];
    stored.updateHistory.unshift({ hash: latestUpdate.versionHash, timestamp: latestUpdate.timestamp, date: new Date(latestUpdate.timestamp).toISOString() });
    if (stored.updateHistory.length > 10) stored.updateHistory.pop();
    writeDB('robloxState', stored);
    await sendRobloxUpdate(client, latestUpdate);
    console.log(`✅ New Roblox update detected: ${latestUpdate.versionHash} from ${latestUpdate.source}`);
  } catch (err) { console.error('❌ checkRoblox error:', err); }
}

// Prefix & message handler
client.on('messageCreate', async msg => {
  if (msg.author.bot || !msg.guild) return;
  await handlePrefixCommand(msg);
  await handleAntiSpam(msg);
  await handleAntiAbuse(msg);
  const exempt = msg.member?.permissions.has(PermissionFlagsBits.ModerateMembers) || msg.member?.permissions.has(PermissionFlagsBits.Administrator);
  if (!exempt) {
    const r = filterCheck(msg.content, msg.author.id);
    if (r.matched) {
      await msg.delete().catch(() => { });
      const fcfg = readDB('wordfilter');
      const warnMsg = fcfg.warnMessage || '🚫 {user}, your message was removed.';
      const n = await msg.channel.send({ embeds: [new EmbedBuilder().setColor(0xff4444).setDescription(warnMsg.replace('{user}', msg.author.toString())).setFooter({ text: readDB('config').antibypass !== false ? 'Bypass attempts are also caught.' : '' })] });
      setTimeout(() => n.delete().catch(() => { }), 5000);
      if (fcfg.logChannelId) {
        const lch = msg.guild.channels.cache.get(fcfg.logChannelId);
        if (lch) await lch.send({ embeds: [new EmbedBuilder().setColor(0xff8800).setTitle('🔍 Filter Violation').addFields({ name: 'User', value: `${msg.author} (${msg.author.id})`, inline: true }, { name: 'Channel', value: `${msg.channel}`, inline: true }, { name: 'Word', value: `\`${r.word[0]}${'*'.repeat(Math.max(0, r.word.length - 2))}${r.word[r.word.length - 1] || ''}\``, inline: true }, { name: 'Message', value: msg.content.substring(0, 200) }).setTimestamp()] }).catch(() => { });
      }
      const wordEntry = fcfg.wordActions?.[r.word.toLowerCase()];
      if (wordEntry === 'mute' || wordEntry === 'mute10') await msg.member?.timeout(10 * 60 * 1000, 'Filter violation').catch(() => { });
      try { await msg.author.send({ embeds: [new EmbedBuilder().setColor(0xff4444).setTitle('🚫 Message Removed').setDescription(`Your message in **${msg.guild.name}** was removed for containing a filtered word.`).setTimestamp()] }); } catch { }
    }
  }
});

function logCase(guildId, action, userId, modId, reason, extra = {}) {
  const cases = readDB('cases'); if (!cases[guildId]) cases[guildId] = [];
  const id = cases[guildId].length + 1;
  cases[guildId].push({ id, action, userId, modId, reason, timestamp: new Date().toISOString(), ...extra });
  writeDB('cases', cases);
  return id;
}

'heads' : 'tails', won = choice === result; addCoins(interaction.user.id, won ? bet : -bet); return interaction.reply({ embeds: [new EmbedBuilder().setColor(won ? 0x00d26a : 0xff4444).setTitle('🪙 Coin Flip').setDescription(`You chose **${choice}** — landed on **${result}**!\n${won ? `🎉 Won ${fmtC(bet)}!` : `💸 Lost ${fmtC(bet)}.`}`).addFields({ name: 'Balance', value: fmtC(getEcon(interaction.user.id).coins), inline: true })] }); }
      if (sub === 'dice') { const pred = interaction.options.getString('pred'), exact = interaction.options.getInteger('num'), roll = Math.floor(Math.random() * 6) + 1; let won = false, mult = 0; if (pred === 'low' && roll >= 4) { won = true; mult = 1.9; } if (pred === 'high' && roll <= 3) { won = true; mult = 1.9; } if (pred === 'exact' && roll === exact) { won = true; mult = 5; } const payout = won ? Math.floor(bet * mult) : 0; addCoins(interaction.user.id, won ? payout - bet : -bet); return interaction.reply({ embeds: [new EmbedBuilder().setColor(won ? 0x00d26a : 0xff4444).setTitle('🎲 Dice Roll').setDescription(`Rolled **${roll}**!\n${won ? `🎉 ${mult}× — Won ${fmtC(payout - bet)}!` : `💸 Lost ${fmtC(bet)}.`}`).addFields({ name: 'Balance', value: fmtC(getEcon(interaction.user.id).coins), inline: true })] }); }
      if (sub === 'roulette') { const choice = interaction.options.getString('choice'), num = Math.floor(Math.random() * 37), res = resolveRoulette(choice, num, bet); addCoins(interaction.user.id, res.won ? res.payout - bet : -bet); return interaction.reply({ embeds: [new EmbedBuilder().setColor(res.won ? 0x00d26a : 0xff4444).setTitle('🎡 Roulette').setDescription(`Ball landed on **${res.color} ${res.num}**!\n${res.won ? `🎉 Won ${fmtC(res.payout - bet)}!` : `💸 Lost ${fmtC(bet)}.`}`).addFields({ name: 'Your Bet', value: choice, inline: true }, { name: 'Balance', value: fmtC(getEcon(interaction.user.id).coins), inline: true })] }); }
      if (sub === 'bj') {
        if (bjGames.has(interaction.user.id)) return interaction.reply({ content: '❌ You already have an active game. Finish it first.', ephemeral: true });
        if (e.coins < bet) return interaction.reply({ content: '❌ Not enough coins.', ephemeral: true });
        addCoins(interaction.user.id, -bet);
        const deck = buildDeck(), player = [deck.pop(), deck.pop()], dealer = [deck.pop(), deck.pop()];
        const pv = hval(player);
        if (pv === 21) { addCoins(interaction.user.id, Math.floor(bet * 2.5)); return interaction.reply({ embeds: [bjEmbed({ player, dealer, bet, deck, userId: interaction.user.id, balance: getEcon(interaction.user.id).coins }, 'bj')] }); }
        const game = { player, dealer, deck, bet, userId: interaction.user.id, balance: getEcon(interaction.user.id).coins };
        const msg = await interaction.reply({ embeds: [bjEmbed(game)], components: [bjRow()], fetchReply: true });
        game.messageId = msg.id; bjGames.set(interaction.user.id, game);
        setTimeout(() => { if (bjGames.has(interaction.user.id)) { bjGames.delete(interaction.user.id); } }, 5 * 60 * 1000);
        return;
      }
      if (sub === 'rob') {
        const target = interaction.options.getUser('user'); if (target.id === interaction.user.id) return interaction.reply({ content: '❌ Cannot rob yourself.', ephemeral: true });
        const te = getEcon(interaction.user.id), now = Date.now(), cd = ROB_CD - (now - te.lastRob);
        if (cd > 0) return interaction.reply({ content: `⏰ Rob on cooldown — **${cdFmt(cd)}** remaining.`, ephemeral: true });
        te.lastRob = now; setEcon(interaction.user.id, te);
        const victim = getEcon(target.id); if (victim.coins < 50) return interaction.reply({ content: `❌ ${target.username} is too broke (< 50 coins).`, ephemeral: true });
        const hasShield = (victim.inventory?.shield || 0) > 0, success = Math.random() < 0.45;
        if (success) { const stolen = Math.floor(victim.coins * (hasShield ? 0.10 : 0.20)); addCoins(target.id, -stolen); addCoins(interaction.user.id, stolen); return interaction.reply({ embeds: [new EmbedBuilder().setColor(0xff8800).setTitle('🦹 Rob Successful!').setDescription(`Robbed ${target} for ${fmtC(stolen)}!${hasShield ? ' *(Shield reduced loot by 50%)*' : ''}`).addFields({ name: 'Balance', value: fmtC(getEcon(interaction.user.id).coins), inline: true })] }); }
        else { const fine = Math.floor(te.coins * 0.10); addCoins(interaction.user.id, -fine); return interaction.reply({ embeds: [new EmbedBuilder().setColor(0xff4444).setTitle('🚔 Rob Failed!').setDescription(`Caught trying to rob ${target}! Fined ${fmtC(fine)}.`).addFields({ name: 'Balance', value: fmtC(getEcon(interaction.user.id).coins), inline: true })] }); }
      }
    }
    // ----- HEIST (heist) -----
    if (cmd === 'heist') {
      const sub = interaction.options.getSubcommand();
      if (sub === 'start') {
        if (heists.get(interaction.guildId)?.status === 'joining') return interaction.reply({ content: '❌ A heist is already forming! Use `/heist join`.', ephemeral: true });
        const bet = interaction.options.getInteger('bet'), e = getEcon(interaction.user.id);
        if (e.coins < bet) return interaction.reply({ content: `❌ Need ${fmtC(bet)} to start.`, ephemeral: true });
        addCoins(interaction.user.id, -bet);
        heists.set(interaction.guildId, { leaderId: interaction.user.id, minBet: bet, participants: [{ id: interaction.user.id, bet }], channelId: interaction.channel.id, status: 'joining', startTime: Date.now() });
        const joinRow = new ActionRowBuilder().addComponents(new ButtonBuilder().setCustomId('heist_join').setLabel('Join Heist').setStyle(ButtonStyle.Success).setEmoji('🦹'));
        const msg = await interaction.reply({ embeds: [new EmbedBuilder().setColor(0xff8800).setTitle('🏦 Heist Forming!').setDescription(`${interaction.user} is planning a heist! **60 seconds to join.**\nMin buy-in: ${fmtC(bet)}`).addFields({ name: 'Crew', value: `${interaction.user}` }, { name: 'Time Left', value: '60 seconds', inline: true }).setFooter({ text: 'Join with the button or /heist join' }).setTimestamp()], components: [joinRow], fetchReply: true });
        setTimeout(async () => {
          const h = heists.get(interaction.guildId); if (!h || h.status !== 'joining') return;
          h.status = 'active';
          await msg.edit({ embeds: [EmbedBuilder.from(msg.embeds[0]).setDescription(`Heist underway! 🔫\n\n**Crew (${h.participants.length}):** ${h.participants.map(p => `<@${p.id}>`).join(', ')}`).setColor(0xff4444)], components: [] }).catch(() => { });
          setTimeout(() => resolveHeist(client, interaction.guildId, h.channelId), 5000);
        }, 60000);
        return;
      }
      if (sub === 'join') {
        const h = heists.get(interaction.guildId); if (!h || h.status !== 'joining') return interaction.reply({ content: '❌ No active heist lobby.', ephemeral: true });
        if (h.participants.find(p => p.id === interaction.user.id)) return interaction.reply({ content: '❌ Already in.', ephemeral: true });
        const bet = interaction.options.getInteger('bet'); if (bet < h.minBet) return interaction.reply({ content: `❌ Min bet is ${fmtC(h.minBet)}.`, ephemeral: true });
        const e = getEcon(interaction.user.id); if (e.coins < bet) return interaction.reply({ content: `❌ Need ${fmtC(bet)}.`, ephemeral: true });
        addCoins(interaction.user.id, -bet); h.participants.push({ id: interaction.user.id, bet });
        return interaction.reply({ embeds: [new EmbedBuilder().setColor(0x00d26a).setTitle('✅ Joined the Heist!').setDescription(`Crew: ${h.participants.map(p => `<@${p.id}>`).join(', ')}\nYour bet: ${fmtC(bet)}`).setTimestamp()] });
      }
      if (sub === 'status') {
        const h = heists.get(interaction.guildId); if (!h) return interaction.reply({ content: '❌ No active heist.', ephemeral: true });
        const elapsed = Math.floor((Date.now() - h.startTime) / 1000);
        return interaction.reply({ embeds: [new EmbedBuilder().setColor(0xff8800).setTitle('🏦 Active Heist').addFields({ name: 'Status', value: h.status === 'joining' ? `🟡 Forming (${Math.max(0, 60 - elapsed)}s left)` : '🔴 In Progress', inline: true }, { name: 'Crew', value: `${h.participants.length}`, inline: true }, { name: 'Pot', value: fmtC(h.participants.reduce((a, p) => a + p.bet, 0)), inline: true }, { name: 'Members', value: h.participants.map(p => `<@${p.id}> — ${fmtC(p.bet)}`).join('\n') }).setTimestamp()], ephemeral: true });
      }
    }
  } catch (err) {
    console.error(`Error in /${cmd}:`, err);
    const msg = { content: `❌ An error occurred: ${err.message}`, ephemeral: true };
    if (interaction.replied || interaction.deferred) await interaction.followUp(msg).catch(() => { });
    else await interaction.reply(msg).catch(() => { });
  }
});

client.login(process.env.BOT_TOKEN);
